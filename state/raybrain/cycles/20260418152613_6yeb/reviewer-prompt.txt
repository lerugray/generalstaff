+"""Ragas eval runner — Phase 1 plumbing.
+
+Loads a golden-set YAML, calls a retrieval function per query, scores each
+(query, retrieved, expected) tuple, aggregates per-query + mean scores.
+Tests inject a stub scorer; production uses the ragas-backed default.
+"""
+from __future__ import annotations
+
+from collections.abc import Callable, Mapping
+from dataclasses import dataclass, field
+from pathlib import Path
+from statistics import fmean
+from typing import Any
+
+import yaml
+
+RetrievalFn = Callable[[str], list[dict[str, Any]]]
+ScorerFn = Callable[[str, list[dict[str, Any]], list[dict[str, Any]]], dict[str, float]]
+
+
+class GoldenSetError(ValueError):
+    """Golden-set YAML missing, unreadable, or malformed."""
+
+
+@dataclass(frozen=True)
+class QueryScore:
+    query: str
+    retrieved: list[dict[str, Any]]
+    expected: list[dict[str, Any]]
+    scores: dict[str, float]
+
+
+@dataclass(frozen=True)
+class EvalReport:
+    per_query: list[QueryScore]
+    mean_scores: dict[str, float] = field(default_factory=dict)
+
+
+def load_golden_set(path: Path) -> list[dict[str, Any]]:
+    if not path.exists():
+        raise GoldenSetError(
+            f"Golden set not found at {path}. See src/raybrain/eval/README.md."
+        )
+    try:
+        raw = yaml.safe_load(path.read_text(encoding="utf-8"))
+    except yaml.YAMLError as exc:
+        raise GoldenSetError(f"Invalid YAML at {path}: {exc}") from exc
+    if raw is None:
+        return []
+    if not isinstance(raw, list):
+        raise GoldenSetError(f"Golden set at {path} must be a YAML list.")
+    for i, entry in enumerate(raw):
+        if not isinstance(entry, Mapping) or not isinstance(entry.get("query"), str):
+            raise GoldenSetError(f"Entry #{i} at {path} missing string 'query'.")
+    return [dict(e) for e in raw]
+
+
+def run_eval(
+    golden_set_path: Path, retrieval_fn: RetrievalFn, scorer: ScorerFn | None = None,
+) -> EvalReport:
+    entries = load_golden_set(golden_set_path)
+    scorer = scorer or _default_ragas_scorer()
+    per_query: list[QueryScore] = []
+    buckets: dict[str, list[float]] = {}
+    for entry in entries:
+        query = entry["query"]
+        expected = list(entry.get("expected_sources", []) or [])
+        retrieved = list(retrieval_fn(query))
+        scores = dict(scorer(query, retrieved, expected))
+        per_query.append(QueryScore(query, retrieved, expected, scores))
+        for name, value in scores.items():
+            buckets.setdefault(name, []).append(float(value))
+    return EvalReport(per_query, {n: fmean(v) for n, v in buckets.items() if v})
+
+
+def _default_ragas_scorer() -> ScorerFn:
+    def _scorer(query, retrieved, expected):
+        from ragas.dataset_schema import SingleTurnSample
+        from ragas.metrics import context_precision, context_recall
+
+        sample = SingleTurnSample(
+            user_input=query,
+            retrieved_contexts=[_passage_text(p) for p in retrieved],
+            reference=" ".join(_passage_text(p) for p in expected) or None,
+        )
+        return {
+            "context_precision": float(context_precision.single_turn_score(sample)),
+            "context_recall": float(context_recall.single_turn_score(sample)),
+        }
+    return _scorer
+
+
+def _passage_text(passage: Mapping[str, Any]) -> str:
+    for key in ("text", "content", "body", "passage"):
+        value = passage.get(key)
+        if isinstance(value, str) and value:
+            return value
+    return ""

+"""Render an EvalReport to markdown."""
+from __future__ import annotations
+
+from pathlib import Path
+
+from raybrain.eval.ragas_runner import EvalReport
+
+
+def render_report(report: EvalReport, out_path: Path | None = None) -> str:
+    lines: list[str] = ["# raybrain eval report", ""]
+    if report.mean_scores:
+        lines += ["## Mean scores", ""]
+        for m in sorted(report.mean_scores):
+            lines.append(f"- **{m}**: {report.mean_scores[m]:.4f}")
+        lines.append("")
+    lines += ["## Per-query results", ""]
+    if not report.per_query:
+        lines.append("_No queries in golden set._")
+    for i, q in enumerate(report.per_query, start=1):
+        lines += [f"### {i}. {q.query}", "",
+                  f"- retrieved: {len(q.retrieved)}", f"- expected: {len(q.expected)}"]
+        for m in sorted(q.scores):
+            lines.append(f"- {m}: {q.scores[m]:.4f}")
+        lines.append("")
+    text = "\n".join(lines).rstrip() + "\n"
+    if out_path is not None:
+        out_path.write_text(text, encoding="utf-8")
+    return text
